289 template<typename T>
290 std::vector<T> Adjacency_List<T>::vertexNeighborhood(const T &p_v, bool p_closed) const {
291  if (!hasVertex(p_v)) {
292  throw logic_error("vertexNeighborhood: the vertex isn't in the graph");
293  }
294  vector<T> adjs;
295  unsigned int index_v = _index(p_v);
296 
297  for (unsigned int pos = 0; pos < m_nodes.size(); pos++) {
298  Node *cur = m_nodes[pos];
299 
300  if (pos == index_v) {
301  // if we find the node of the given vertex, we add to the vector
302  // all the vertices it points to, after having checked they're not already in it
303  for (typename list<T>::const_iterator vert =
304  cur->m_neighbors.begin(); vert != cur->m_neighbors.end();
305  ++vert) {
306  if (std::find(adjs.begin(), adjs.end(), (*vert))
307  == adjs.end()) {
308  adjs.push_back((*vert));
309  }
310  }
311  } else {
312  // for any other vertex, we check if the given vertex is one of the neighbors.
313  // if it is, add this vertex to the vector after having checked it isn't already in it
314  if (std::find(cur->m_neighbors.begin(), cur->m_neighbors.end(),
315  m_nodes[index_v]->m_data) != cur->m_neighbors.end()) {
316  if (std::find(adjs.begin(), adjs.end(), cur->m_data)
317  == adjs.end()) {
318  adjs.push_back(cur->m_data);
319  }
320  }
321  }
322  }
323  if (p_closed) {
324  adjs.push_back(p_v);
325  }
326  return adjs;
327 }
